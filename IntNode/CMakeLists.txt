cmake_minimum_required(VERSION 3.10)

# Set the project name
project(IntList)

# Set the C++ standard
set(CMAKE_CXX_STANDARD 17)

# Include directories (add the path to your CxxTest installation)
include_directories(/opt/homebrew/Cellar/cxxtest/4.4_3/include/)

# Set the path to the cxxtestgen executable
set(CXXTEST_PYTHON_TESTGEN_EXECUTABLE /opt/homebrew/Cellar/cxxtest/4.4_3/bin/cxxtestgen)

# Find CxxTest package
find_package(CxxTest REQUIRED)

# Enable testing
enable_testing()


# Generate the runner.cpp and create the test executable
CXXTEST_ADD_TEST(unittest                      # Name of the test target
        runner.cpp                                # This file will be generated by cxxtestgen
        ${CMAKE_CURRENT_SOURCE_DIR}/IntNode.cpp  # Source file for your project
        ${CMAKE_CURRENT_SOURCE_DIR}/IntList.cpp  # Source file for your project
        ${CMAKE_CURRENT_SOURCE_DIR}/newCxxTest.h  # CxxTest description file (your test suite)
)

# Add the executable for the restaurant project (make sure to include the source files)
add_executable(IntList
        runner.cpp               # Generated by cxxtestgen
        IntNode.cpp             # Source file
        IntList.cpp             # Source file
        newCxxTest.h             # CxxTest description file
)
